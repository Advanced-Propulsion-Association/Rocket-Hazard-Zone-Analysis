# 6-DOF V2 Hazard Zone Calculator Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a standalone 6-DOF hobby rocket hazard zone calculator (v2) with Monte Carlo dispersion, attitude history, and 3-DOF comparison panel, living alongside v1 in the same repo.

**Architecture:** Fork v1 into `hazard-zone-calculator-v2/`. Replace the 3-DOF integrator with a 13-state RK4 6-DOF engine. Run N dispersion simulations in a Web Worker. Primary input is an OpenRocket `.ork` file; manual entry is the fallback. Side-by-side UI: input left, results right.

**Tech Stack:** React 19, TypeScript, Vite 8, Tailwind CSS v4, Plotly.js, Vitest, native Web Worker API.

```typescript
/**
 * Barrowman aerodynamic coefficient calculator.
 *
 * Computes the normal force coefficient slope CNα and the center of pressure
 * location using the simplified Barrowman equations (1967).
 *
 * Reference: Barrowman, J.S., "The Practical Calculation of the Aerodynamic
 * Characteristics of Slender Finned Vehicles", MS Thesis, Catholic University, 1967.
 */

export interface BarrowmanInputs {
  bodyDiameter_m: number;
  noseLength_m: number;
  noseConeType: 'ogive' | 'conical' | 'parabolic' | 'haack';
  finRootChord_m: number;
  finTipChord_m: number;
  finSpan_m: number;
  numFins: number;
}

export interface BarrowmanInputsWithBody extends BarrowmanInputs {
  bodyLength_m: number;
}

/**
 * Normal force coefficient slope (per radian).
 * CNα = CNα_nose + CNα_fins
 */
export function computeCNAlpha(inputs: BarrowmanInputs): number {
  const { bodyDiameter_m, finRootChord_m, finTipChord_m, finSpan_m, numFins } = inputs;
  const R = bodyDiameter_m / 2;

  // Nose contribution: slender body theory gives CNα = 2 for all nose shapes
  const CNalpha_nose = 2.0;

  if (numFins === 0 || finSpan_m <= 0 || finRootChord_m <= 0) {
    return CNalpha_nose;
  }

  // Fin planform area (trapezoidal fin)
  const Af = (finRootChord_m + finTipChord_m) * finSpan_m / 2;

  // Effective aspect ratio of fin panel (not full span)
  const AR = finSpan_m * finSpan_m / Af;

  // Barrowman fin CNα per fin (lifting surface theory)
  const CNalpha_perFin = (2 * Math.PI * AR) / (2 * (1 + Math.sqrt(1 + Math.pow(AR / 2, 2))));

  // Body-fin interference factor: accounts for body blocking fin root
  const Kfb = 1 + R / (R + finSpan_m);

  const CNalpha_fins = numFins * CNalpha_perFin * Kfb;

  return CNalpha_nose + CNalpha_fins;
}

/**
 * Center of pressure location measured from nose tip (meters).
 */
export function computeCPFromNose(inputs: BarrowmanInputsWithBody): number {
  const {
    bodyDiameter_m, noseLength_m, bodyLength_m,
    finRootChord_m, finTipChord_m, finSpan_m, numFins,
  } = inputs;
  const R = bodyDiameter_m / 2;

  // Nose CP: at ~46% of nose length for ogive/parabolic, at 67% for conical
  const noseCPFraction = inputs.noseConeType === 'conical' ? 2 / 3 : 0.466;
  const xCP_nose = noseLength_m * noseCPFraction;
  const CNalpha_nose = 2.0;

  if (numFins === 0 || finSpan_m <= 0 || finRootChord_m <= 0) {
    return xCP_nose;
  }

  const Af = (finRootChord_m + finTipChord_m) * finSpan_m / 2;
  const AR = finSpan_m * finSpan_m / Af;
  const CNalpha_perFin = (2 * Math.PI * AR) / (2 * (1 + Math.sqrt(1 + Math.pow(AR / 2, 2))));
  const Kfb = 1 + R / (R + finSpan_m);
  const CNalpha_fins = numFins * CNalpha_perFin * Kfb;

  // Fin CP: at the centroid of the fin planform, measured from nose
  // Fin leading edge is at body length minus root chord (trailing edge at body end)
  const xFinLeadingEdge = bodyLength_m - finRootChord_m;
  // Centroid of trapezoid along root chord: (r + 2t)/(3(r+t)) * root_chord
  const xCP_fins_local = finRootChord_m *
    (finRootChord_m + 2 * finTipChord_m) / (3 * (finRootChord_m + finTipChord_m));
  const xCP_fins = xFinLeadingEdge + xCP_fins_local;

  // Area-weighted average CP location
  const CNalpha_total = CNalpha_nose + CNalpha_fins;
  return (CNalpha_nose * xCP_nose + CNalpha_fins * xCP_fins) / CNalpha_total;
}
```

```typescript
/**
 * Moment of inertia estimation for hobby rockets.
 *
 * Uses a uniform-density solid cylinder approximation. This is conservative:
 * real rockets with heavy motors aft have higher Iyy than this estimate,
 * which increases stability — so the approximation is slightly pessimistic.
 */

export interface MOIInputs {
  totalMass_kg: number;
  bodyDiameter_m: number;
  totalLength_m: number;
}

export interface MOIResult {
  Ixx: number;   // axial (roll) MOI in kg·m²
  Iyy: number;   // transverse (pitch/yaw) MOI in kg·m²
}

/**
 * Solid cylinder MOI approximation.
 *   Ixx = 0.5 * m * R²           (spin about long axis)
 *   Iyy = m * (3R² + L²) / 12   (spin about transverse axis through CG)
 */
export function estimateMOI(inputs: MOIInputs): MOIResult {
  const { totalMass_kg: m, bodyDiameter_m, totalLength_m: L } = inputs;
  const R = bodyDiameter_m / 2;
  return {
    Ixx: 0.5 * m * R * R,
    Iyy: m * (3 * R * R + L * L) / 12,
  };
}

/**
 * Estimate pitch/yaw damping coefficient Cmq (= Cnr for symmetric rocket).
 * Empirical approximation: Cmq ≈ -2 * CNα * (L/2) / d
 * where L = reference length, d = body diameter.
 */
export function estimateCmq(CNalpha: number, bodyLength_m: number, bodyDiameter_m: number): number {
  return -2 * CNalpha * (bodyLength_m / 2) / bodyDiameter_m;
}

/**
 * Estimate roll damping coefficient Clp.
 * Empirical: Clp ≈ -4 * (fin planform area) * (fin moment arm)² / (body diameter * ref area * L)
 * Simplified to: Clp ≈ -0.5 (typical for 3–4 fin hobby rocket)
 */
export function estimateClp(): number {
  return -0.5;
}
```

```typescript
/**
 * Extract number of fins from parsed .ork document.
 * Falls back to 3 if not found (most common hobby rocket config).
 */
export function extractNumFins(doc: Document): number {
  const finEl =
    doc.getElementsByTagName('trapezoidfinset')[0] ??
    doc.getElementsByTagName('ellipticalfinset')[0] ??
```

```typescript
/**
 * 6-DOF point-mass trajectory integrator (RK4).
 *
 * Coordinate system (inertial frame):
 *   x = downrange (aligned with wind direction, +x downwind)
 *   y = lateral (crosswind)
 *   z = altitude AGL (positive up)
 *
 * Euler angle convention:
 *   theta = pitch angle from vertical (0 = straight up, π/2 = horizontal)
 *   psi   = yaw angle (azimuth of nose, 0 = into wind = along +x)
 *   phi   = roll angle about body axis
 *
 * NOTE: Kinematic equations have a singularity at theta = ±π/2. Hobby rockets
 * launched within 20° of vertical never approach this. A guard clamps cos(theta)
 * to ±0.01 to avoid division by zero if simulation diverges.
 *
 * PHYSICS REVIEW REQUIRED before regulatory use: verify Barrowman moment
 * arm signs, damping coefficient directions, and gyroscopic terms.
 */

import type { Config6DOF, TrajectoryPoint6DOF } from '../types';
import { G0, airDensity, speedOfSound, isaTemperatureOffset, isaPressure } from './atmosphere';
import { cdFromFineness, cdMachCorrection } from './aerodynamics';
import { thrustAt, totalImpulse } from './motor';

export interface SimOptions6DOF {
  initialZ_m?: number;   // starting altitude AGL (default 0)
  dtMax?: number;
}

// State and derivative share the same shape (13 numbers)
type S = {
  x: number; y: number; z: number;
  vx: number; vy: number; vz: number;
  phi: number; theta: number; psi: number;
  p: number; q: number; r: number;
  mass: number;
};

function addS(a: S, b: S, dt: number): S {
  return {
    x: a.x + b.x * dt, y: a.y + b.y * dt, z: a.z + b.z * dt,
    vx: a.vx + b.vx * dt, vy: a.vy + b.vy * dt, vz: a.vz + b.vz * dt,
    phi: a.phi + b.phi * dt, theta: a.theta + b.theta * dt, psi: a.psi + b.psi * dt,
    p: a.p + b.p * dt, q: a.q + b.q * dt, r: a.r + b.r * dt,
    mass: a.mass + b.mass * dt,
  };
}

function midS(a: S, b: S, dt: number): S { return addS(a, b, dt / 2); }

function derivs(t: number, s: S, cfg: Config6DOF, dryMass: number, I_total: number, tOffset: number): S {
  const { x: _x, y: _y, z, vx, vy, vz, phi, theta, psi, p, q, r, mass } = s;
  void _x; void _y;

  const absZ = Math.max(z + cfg.siteElevation_m, 0);
  const rho = airDensity(absZ, tOffset);
  const aSnd = speedOfSound(absZ, tOffset);
  const P = isaPressure(absZ);
  const A_ref = Math.PI * (cfg.bodyDiameter_m / 2) ** 2;
  const fineness = cfg.bodyLength_m / cfg.bodyDiameter_m;
  const cdSub = cdFromFineness(fineness);

  const T_raw = thrustAt(cfg.motor, t);
  const T = cfg.motor.nozzleExitAreaM2
    ? T_raw + (101325 - P) * cfg.motor.nozzleExitAreaM2
    : T_raw;

  // Body axis (nose direction) in inertial frame
  const sinTheta = Math.sin(theta), cosTheta = Math.cos(theta);
  const sinPsi = Math.sin(psi), cosPsi = Math.cos(psi);
  const nx = sinTheta * cosPsi;
  const ny = sinTheta * sinPsi;
  const nz = cosTheta;

  // Relative velocity (subtract wind along +x)
  const vxR = vx - cfg.surfaceWind_ms, vyR = vy, vzR = vz;
  const vRel = Math.sqrt(vxR ** 2 + vyR ** 2 + vzR ** 2);

  // Forces (start with gravity)
  let Fx = 0, Fy = 0, Fz = -mass * G0;
  // Moments (body frame: L=roll, M=pitch, N=yaw)
  let Lm = 0, Mm = 0, Nm = 0;

  if (vRel > 0.1) {
    const mach = vRel / aSnd;
    const cd = cdMachCorrection(cdSub, mach);
    const q_dyn = 0.5 * rho * vRel ** 2;

    // Drag
    const D = q_dyn * A_ref * cd;
    Fx -= D * vxR / vRel;
    Fy -= D * vyR / vRel;
    Fz -= D * vzR / vRel;

    // Angle of attack
    const vxRh = vxR / vRel, vyRh = vyR / vRel, vzRh = vzR / vRel;
    const dotNV = nx * vxRh + ny * vyRh + nz * vzRh;
    const alpha = Math.acos(Math.max(-1, Math.min(1, dotNV)));

    if (alpha > 0.001) {
      // Normal force — perpendicular to velocity, in (nose, velocity) plane
      const FN = cfg.CNalpha * alpha * q_dyn * A_ref;
      const npx = nx - dotNV * vxRh;
      const npy = ny - dotNV * vyRh;
      const npz = nz - dotNV * vzRh;
      const npMag = Math.sqrt(npx ** 2 + npy ** 2 + npz ** 2);
      if (npMag > 0.001) {
        Fx += FN * npx / npMag;
        Fy += FN * npy / npMag;
        Fz += FN * npz / npMag;
      }

      // Restoring moment: arm = xCP - xCG (positive = stable = nose ahead of CG)
      const arm = cfg.xCP_m - cfg.xCG_m;
      const Mrestore = -FN * arm;
      const sinPhi = Math.sin(phi), cosPhi = Math.cos(phi);
      Mm += Mrestore * cosPhi;
      Nm += Mrestore * sinPhi;
    }

    // Damping moments
    const L_ref = cfg.bodyLength_m;
    const d_ref = cfg.bodyDiameter_m;
    Mm += cfg.Cmq * q * q_dyn * A_ref * L_ref;
    Nm += cfg.Cnr * r * q_dyn * A_ref * L_ref;
    Lm += cfg.Clp * p * q_dyn * A_ref * d_ref;
  }

  // Thrust along body axis
  Fx += T * nx;
  Fy += T * ny;
  Fz += T * nz;

  // Translational accelerations
  const ax = Fx / mass, ay = Fy / mass, az = Fz / mass;

  // Euler rotation equations (body frame, symmetric rocket: Izz = Iyy)
  const dp_dt = Lm / cfg.Ixx;
  const dq_dt = (Mm - (cfg.Iyy - cfg.Ixx) * p * r) / cfg.Iyy;
  const dr_dt = (Nm + (cfg.Iyy - cfg.Ixx) * p * q) / cfg.Iyy;

  // Euler kinematic equations (singularity guard at theta = ±90°)
  const sinPhi = Math.sin(phi), cosPhi = Math.cos(phi);
  const cosThetaGuarded = Math.abs(cosTheta) > 0.01 ? cosTheta : Math.sign(cosTheta) * 0.01;
  const tanTheta = sinTheta / cosThetaGuarded;
  const dphi_dt = p + (q * sinPhi + r * cosPhi) * tanTheta;
  const dtheta_dt = q * cosPhi - r * sinPhi;
  const dpsi_dt = (q * sinPhi + r * cosPhi) / cosThetaGuarded;

  // Mass flow
  let dm_dt = 0;
  if (I_total > 0 && T_raw > 0 && mass > dryMass) {
    dm_dt = -cfg.motor.propellantMassKg * T_raw / I_total;
  }

  return {
    x: vx, y: vy, z: vz,
    vx: ax, vy: ay, vz: az,
    phi: dphi_dt, theta: dtheta_dt, psi: dpsi_dt,
    p: dp_dt, q: dq_dt, r: dr_dt,
    mass: dm_dt,
  };
}

export function simulate6DOF(
  cfg: Config6DOF & { initialZ_m?: number },
  opts: SimOptions6DOF = {},
): TrajectoryPoint6DOF[] {
  const dtMax = opts.dtMax ?? 0.05;
  const I_total = totalImpulse(cfg.motor);
  const dryMass = cfg.totalMass_kg - cfg.motor.propellantMassKg;
  const siteElev = cfg.siteElevation_m;
  const tOffset = isaTemperatureOffset(siteElev, cfg.siteTemp_K);

  const z0 = cfg.initialZ_m ?? 0;
  const theta0 = (cfg.launchAngle_deg * Math.PI) / 180;
  const psi0 = (cfg.launchAzimuth_deg * Math.PI) / 180;

  let s: S = {
    x: 0, y: 0, z: z0,
    vx: 0, vy: 0, vz: 0,
    phi: 0, theta: theta0, psi: psi0,
    p: cfg.initialRollRate_rads, q: cfg.initialQ_rads ?? 0, r: 0,
    mass: cfg.totalMass_kg,
  };

  const points: TrajectoryPoint6DOF[] = [];
  let t = 0;
  let wasAboveGround = z0 > 5;

  while (t < 600) {
    if (s.z > 5) wasAboveGround = true;
    if (wasAboveGround && s.z < -1.0) break;
    if (!wasAboveGround && t > 30) break;

    // Record point
    {
      const absZ = Math.max(s.z + siteElev, 0);
      const vxR = s.vx - cfg.surfaceWind_ms;
      const vRel = Math.sqrt(vxR ** 2 + s.vy ** 2 + s.vz ** 2);
      const aSnd = speedOfSound(absZ, tOffset);
      const mach = vRel / (aSnd || 1);
      const cdSub = cdFromFineness(cfg.bodyLength_m / cfg.bodyDiameter_m);
      const cd = cdMachCorrection(cdSub, mach);
      const rho = airDensity(absZ, tOffset);
      const A_ref = Math.PI * (cfg.bodyDiameter_m / 2) ** 2;
      const D = 0.5 * rho * vRel ** 2 * A_ref * cd;
      const T = thrustAt(cfg.motor, t);
      const sinTheta = Math.sin(s.theta), cosTheta = Math.cos(s.theta);
      const nx = sinTheta * Math.cos(s.psi);
      const ny = sinTheta * Math.sin(s.psi);
      const nz = cosTheta;
      const vRh = vRel > 0.1 ? vRel : 1;
      const dotNV = (nx * vxR + ny * s.vy + nz * s.vz) / vRh;
      const alpha = Math.acos(Math.max(-1, Math.min(1, dotNV)));
      points.push({
        t, x: s.x, y: s.y, z: s.z,
        vx: s.vx, vy: s.vy, vz: s.vz,
        phi: s.phi, theta: s.theta, psi: s.psi,
        p: s.p, q: s.q, r: s.r,
        alpha, mass: s.mass, mach, thrust: T, drag: D,
      });
    }

    const dt = thrustAt(cfg.motor, t) > 0 ? 0.01 : Math.min(dtMax, 0.05);
    const k1 = derivs(t,        s,               cfg, dryMass, I_total, tOffset);
    const k2 = derivs(t+dt/2,   midS(s, k1, dt), cfg, dryMass, I_total, tOffset);
    const k3 = derivs(t+dt/2,   midS(s, k2, dt), cfg, dryMass, I_total, tOffset);
    const k4 = derivs(t+dt,     addS(s, k3, dt), cfg, dryMass, I_total, tOffset);

    s = {
      x:     s.x     + (dt/6)*(k1.x     + 2*k2.x     + 2*k3.x     + k4.x),
      y:     s.y     + (dt/6)*(k1.y     + 2*k2.y     + 2*k3.y     + k4.y),
      z:     s.z     + (dt/6)*(k1.z     + 2*k2.z     + 2*k3.z     + k4.z),
      vx:    s.vx    + (dt/6)*(k1.vx    + 2*k2.vx    + 2*k3.vx    + k4.vx),
      vy:    s.vy    + (dt/6)*(k1.vy    + 2*k2.vy    + 2*k3.vy    + k4.vy),
      vz:    s.vz    + (dt/6)*(k1.vz    + 2*k2.vz    + 2*k3.vz    + k4.vz),
      phi:   s.phi   + (dt/6)*(k1.phi   + 2*k2.phi   + 2*k3.phi   + k4.phi),
      theta: s.theta + (dt/6)*(k1.theta + 2*k2.theta + 2*k3.theta + k4.theta),
      psi:   s.psi   + (dt/6)*(k1.psi   + 2*k2.psi   + 2*k3.psi   + k4.psi),
      p:     s.p     + (dt/6)*(k1.p     + 2*k2.p     + 2*k3.p     + k4.p),
      q:     s.q     + (dt/6)*(k1.q     + 2*k2.q     + 2*k3.q     + k4.q),
      r:     s.r     + (dt/6)*(k1.r     + 2*k2.r     + 2*k3.r     + k4.r),
      mass:  Math.max(s.mass + (dt/6)*(k1.mass + 2*k2.mass + 2*k3.mass + k4.mass), dryMass),
    };
    t += dt;
  }

  return points;
}
```

```typescript
import type { Config6DOF, MonteCarloResult, ScatterPoint } from '../types';
import { simulate6DOF } from './trajectory6dof';

export interface VarianceOptions {
  windVariance?: number;      // fraction of wind speed (default 0.20 = ±20%)
  angleVariance?: number;     // degrees (default 2)
  impulseVariance?: number;   // fraction of total impulse (default 0.03 = ±3%)
  pitchPerturbance?: number;  // degrees initial pitch (default 1)
}

/** Uniform random in [-1, 1] */
function rand(): number { return Math.random() * 2 - 1; }

/** Compute hazard radius = max distance from launch pad (origin) */
export function hazardRadiusFromPoints(points: ScatterPoint[]): number {
  return Math.max(...points.map(p => Math.sqrt(p.x * p.x + p.y * p.y)));
}

/** 99th-percentile distance from launch pad */
function p99Radius(points: ScatterPoint[]): number {
  const dists = points.map(p => Math.sqrt(p.x * p.x + p.y * p.y)).sort((a, b) => a - b);
  const idx = Math.min(Math.floor(dists.length * 0.99), dists.length - 1);
  return dists[idx];
}

export function runMonteCarlo(
  nominalConfig: Config6DOF,
  numRuns: number,
  onProgress: (completed: number) => void,
  variance: VarianceOptions = {},
): MonteCarloResult {
  const {
    windVariance    = 0.20,
    angleVariance   = 2.0,
    impulseVariance = 0.03,
    pitchPerturbance = 1.0,
  } = variance;

  const points: ScatterPoint[] = [];
  let nominalTrajectory = simulate6DOF({ ...nominalConfig, initialZ_m: 0 });

  for (let i = 0; i < numRuns; i++) {
    // Randomize wind speed and direction
    const windSpeed = nominalConfig.surfaceWind_ms * (1 + windVariance * rand());
    const windAzimuth = Math.random() * 2 * Math.PI; // 0–360° uniform

    // Wind along x-axis in inertial frame — rotate launch into wind frame
    // Simplification: apply wind direction by rotating launch azimuth
    const launchAzimuth = nominalConfig.launchAzimuth_deg + (windAzimuth * 180 / Math.PI);
    const launchAngle = nominalConfig.launchAngle_deg + angleVariance * rand();

    // Motor impulse scaling (scale thrust curve via a wrapper)
    const impulseScale = 1 + impulseVariance * rand();
    const scaledMotor = {
      ...nominalConfig.motor,
      thrustCurve: nominalConfig.motor.thrustCurve.map(pt => ({
        ...pt,
        thrust: pt.thrust * impulseScale,
      })),
    };

    // Initial pitch perturbation
    const pitchPerturb = pitchPerturbance * rand() * Math.PI / 180;

    const runConfig: Config6DOF & { initialZ_m?: number; initialQ_rads?: number } = {
      ...nominalConfig,
      motor: scaledMotor,
      surfaceWind_ms: Math.abs(windSpeed),
      launchAzimuth_deg: launchAzimuth,
      launchAngle_deg: Math.max(0, launchAngle),
      initialQ_rads: pitchPerturb,
    };

    const pts = simulate6DOF(runConfig);

    const last = pts[pts.length - 1];
    points.push({ x: last.x, y: last.y, runIndex: i });

    if (i === 0) nominalTrajectory = pts;
    if ((i + 1) % 50 === 0) onProgress(i + 1);
  }

  onProgress(numRuns);

  const hazardRadius_m = hazardRadiusFromPoints(points);
  const M_TO_FT = 3.28084;

  return {
    points,
    hazardRadius_m,
    hazardRadius_ft: hazardRadius_m * M_TO_FT,
    hazardRadius_p99_m: p99Radius(points),
    nominalTrajectory,
    nominalConfig,
  };
}
```

```typescript
import type { OpenRocketData } from '../types';

interface Props {
  onParsed: (data: OpenRocketData) => void;
  onError: (msg: string) => void;
}

export function OrkUpload({ onParsed, onError }: Props) {
  const [dragging, setDragging] = useState(false);
  const [fileName, setFileName] = useState<string | null>(null);

  const handleFile = useCallback(async (file: File) => {
    if (!file.name.endsWith('.ork')) {
      onError('Please upload a .ork file (OpenRocket format).');
      return;
    }
    try {
      const buffer = await file.arrayBuffer();
      const data = await parseOrkFile(buffer);
      setFileName(file.name);
      onParsed(data);
    } catch (e) {
      onError(`Failed to parse .ork file: ${e instanceof Error ? e.message : String(e)}`);
    }
  }, [onParsed, onError]);

  const onDrop = useCallback((e: React.DragEvent) => {
    e.preventDefault();
    setDragging(false);
    const file = e.dataTransfer.files[0];
    if (file) handleFile(file);
  }, [handleFile]);

  const onInputChange = useCallback((e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (file) handleFile(file);
  }, [handleFile]);

  return (
    <div>
      <label
        className={`flex flex-col items-center justify-center border-2 border-dashed rounded-lg p-6 cursor-pointer transition-colors ${
          dragging ? 'border-blue-400 bg-blue-900/20' : 'border-gray-600 hover:border-blue-500'
        }`}
        onDragOver={e => { e.preventDefault(); setDragging(true); }}
        onDragLeave={() => setDragging(false)}
        onDrop={onDrop}
      >
        <span className="text-2xl mb-2">📂</span>
        <span className="text-sm text-gray-300">
          {fileName ? `Loaded: ${fileName}` : 'Drop .ork file here or click to browse'}
        </span>
        <input type="file" accept=".ork" className="hidden" onChange={onInputChange} />
      </label>
    </div>
  );
```

```typescript
    workerRef.current?.terminate();
    workerRef.current = new Worker(
      new URL('./workers/montecarlo.worker.ts', import.meta.url),
      { type: 'module' }
    );

    workerRef.current.onmessage = (e: MessageEvent<WorkerProgress | WorkerResult>) => {
      if (e.data.type === 'progress') {
        setProgress(e.data.completed / e.data.total);
      } else if (e.data.type === 'result') {
        setResult(e.data.result);
        setRunning(false);
        setProgress(1);
      }
    };

    workerRef.current.onerror = (e) => {
      setError(`Simulation error: ${e.message}`);
      setRunning(false);
    };

    workerRef.current.postMessage({ config, numRuns });
  }, [orkData, manualData, numRuns]);

  const effectiveData = orkData ?? manualData;

  return (
    <div className="min-h-screen bg-gray-950 text-white">
      <header className="bg-gray-900 border-b border-gray-800 px-6 py-3">
        <h1 className="text-lg font-semibold">FAA Hazard Zone Calculator <span className="text-blue-400 text-sm font-normal ml-2">v2 — 6-DOF</span></h1>
      </header>

      <div className="flex h-[calc(100vh-52px)]">
        {/* Left panel — inputs */}
        <div className="w-[420px] flex-shrink-0 bg-gray-900 border-r border-gray-800 overflow-y-auto p-5">
          <p className="text-xs text-gray-400 uppercase tracking-wider mb-3">Rocket</p>
          <OrkUpload onParsed={data => { setOrkData(data); setError(null); }} onError={setError} />
          <ManualInputForm values={manualData} onChange={d => setManualData(prev => ({ ...prev, ...d }))} />

          {error && (
            <div className="mt-3 p-3 bg-red-900/30 border border-red-700 rounded text-xs text-red-300">
              {error}
            </div>
          )}

          <RunControls
            numRuns={numRuns}
            onNumRunsChange={setNumRuns}
            onRun={handleRun}
            running={running}
            progress={progress}
            canRun={!!(effectiveData as OpenRocketData)?.bodyDiameter_in}
          />
        </div>

        {/* Right panel — results */}
        <div className="flex-1 overflow-y-auto p-5">
          {result ? (
            <Results6dof result={result} />
          ) : (
            <div className="flex items-center justify-center h-full text-gray-600">
              <div className="text-center">
                <p className="text-4xl mb-3">🚀</p>
                <p>Upload a .ork file and run the simulation to see results.</p>
              </div>
            </div>
          )}
        </div>
      </div>
    </div>
  );
}
```
